%=============================================================================%
\begin{frame}{Fees, Loads, and Performance}
\begin{itemize}
\item Gross performance of load funds is statistically identical to gross performance of no-load funds
\item Funds with high expenses tend to be poorer performers
\item 12b-1 charges should be added to expense ratios
\item Compare costs with Morningstar
\end{itemize}
\end{frame}

%=============================================================================%
%   NAV AND EFFECTIVE LOAD                                                    %
%=============================================================================%
\begin{frame}{NAV and Effective Load}
\begin{itemize}
\item Cost to initially purchase one share of load fund = NAV + Front-end load (\%) (if any)
\item Stated loads typically range from 0 to 8.5\%
\item Load is designed to offset expenses of marketing the fund; it goes to broker who sells fund to investor
\item Effective load greater than stated load
\end{itemize}
\end{frame}

%=============================================================================%
%   GENERAL TAX RULES                                                         %
%=============================================================================%
\begin{frame}{General Tax Rules}
\begin{itemize}
\item Fund not taxed if diversified and income distributed
\item Investor taxed on capital gain and dividend distributions
\item Turnover: Ratio of trading activity to assets of portfolio
\item Portfolio turnover may affect investor’s tax liability
\end{itemize}
\end{frame}

%=============================================================================%
%   NAV AND EFFECTIVE LOAD                                                    %
%=============================================================================%
\begin{frame}{Implications of Fund Turnover}
\begin{itemize}
\item Fund pays commission costs on portfolio purchases and sales—charged against NAV
\item Turnover rate measured as annual total asset value bought or sold in a year divided by average total asset value
\end{itemize}
\end{frame}

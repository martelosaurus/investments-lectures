\noindent We want to get to
\begin{equation}
	\mathbb{E}_{t}[R_{i,t+1}-R_{F,t+1}]=\beta_{i}\mathbb{E}_{t}[R_{M,t+1}-R_{F,t+1}]
\end{equation}
where $\beta_{i}=\mathbb{C}_{t}[R_{i,t+1},R_{M,t+1}]/\mathbb{C}_{t}[R_{M,t+1},R_{M,t+1}]$. Given a pricing kernel $M_{t+1}$, 
\begin{equation}
	1=\mathbb{E}_{t}[M_{t+1}R_{F,t+1}]=\mathbb{E}_{t}[M_{t+1}]R_{F,t+1}
\end{equation}
and
\begin{equation}
	1=\mathbb{E}_{t}[M_{t+1}R_{M,t+1}]
\end{equation}

				\draw (4,0) node[above=3pt] {2};
				\draw (8,0) node[above=3pt] {$2T$};
				\draw (0,0) node[below=3pt] {$-P$};
				\draw (2,0) node[below=3pt] {$cF/2$};
				\draw (4,0) node[below=3pt] {$cF/2$};
				\draw (6,0) node[above=3pt, align=center] {$\cdots$};
				\draw (6,0) node[below=3pt, align=center] {$\cdots$};
				\draw (8,0) node[below=3pt] {$cF/2+F$};
			\end{tikzpicture}
		\end{center}
	\end{figure}
where $c\geq0$ is the coupon rate (APR) 
{\scriptsize
\begin{align*}
	P&\should\frac{cF/2}{(1+y/2)}+\frac{cF/2}{(1+y/2)^2}+\frac{cF/2}{(1+y/2)^3}+\cdots+\frac{cF/2}{(1+y/2)^{2T}}+\frac{F}{(1+y/2)^{2T}}\\
	&=\frac{cF/2}{y/2}\left(1-\frac{1}{(1+y/2)^{2T}}\right)+\frac{F}{(1+y/2)^{2T}}
\end{align*}}
where $y$ is the yield (APR).

% Bond Checklist
\begin{frame}{\fttheory{Treasury Checklist}}
	\begin{itemize}
		\item[$\checkmark$] Yields and coupon rates are quoted as APRs, so divide them by the number of payments per year to get the per-period rate
		\item[$\checkmark$] Multiply the number of years by the number of payments per year to get the number of periods
		\item[$\checkmark$] The future value is the face value
		\item[$\checkmark$] \underline{Remember}: if the coupon rate is greater than (less than) the yield, the price is greater than (less than) the face value
		\item[$\checkmark$] \textbf{Use four decimal places!}
	\end{itemize}
\end{frame}

\begin{frame}{\fttheory{Why are there different rates?}}
	\begin{itemize}
		\item Why are there different rates? Why isn't there just one rate for all terms?
		\item When conditions are expected to improve, the curve is upward sloping; when they are expected to deteriorate, the curve is downward sloping
		\item Usually, the curve is upward sloping
		\item When long-term rates (e.g. 20 year) fall below short-term rates (e.g. 1 year), we say that the yield curve has \underline{inverted}
	\end{itemize}
\end{frame}

\begin{frame}{\fttheory{Zero Rates}}
	\begin{itemize}
		\item We can use the yield curve to compute the rates on zero-coupon bonds of various terms called \underline{zero rates} through a process known as \underline{bootstrapping}
		\item $y_{1}$, $y_{2}$ are the 6-and 12-month yields from the yield curve
		\item $z_{1}$, $z_{2}$ are the 6-and 12-month zero rates
		\item The 6-month zero rate is the 6-month yield from the yield curve: $z_{1}=y_{1}$
		\item Solve for $z_{2}$:
			\begin{equation*}
				P=F=\frac{{y_{2}}F/2}{1+{z_{1}}/2}+\frac{{y_{2}}F/2}{(1+{z_{2}/2})^{2}}+\frac{F}{(1+{z_{2}/2})^{2}}
			\end{equation*}
		\item This equation says that $z_{2}$ is the zero rate for a 12-month, semi-annual coupon bond that trades at par
	\end{itemize}
\end{frame}

\begin{frame}{\ftexampl{Zero Rates}}
	\small
	Suppose that $y_{1}=1\%$ and $y_{2}=3\%$. What are the zero rates?\\
	\pause
	{\color{blue}
		$z_{1}=y_{1}=1\%$. Use the following equation for $z_{2}$
		\begin{align*}
		z_{2}&=2\times\text{RATE}(\\
			&\hspace{.5in}\text{N}=2,\\
			&\hspace{.5in}\text{PV}=-(1-(y_{2}/2)/(1+z_{1}/2)),\\
			&\hspace{.5in}\text{PMT}=0,\\
			&\hspace{.5in}\text{FV}=y_{2}/2+1)\\
			&=3.0152\%
		\end{align*}
	}
\end{frame}

	\begin{figure}[H]
		\begin{center}
			\begin{tikzpicture}
				\draw (0,0) -- (8,0);
				\foreach \x in {0,4,8}
				\draw(\x cm,3pt) -- (\x cm, -3pt);

				\draw (4,0) node[above=3pt] {1};
				\draw (8,0) node[above=3pt] {2};
				\draw[->] (0,0) to[bend left] (3.75,.25);
				\draw[->] (4,0) to[bend left] (7.75,.25);
				\draw[->] (0,0) to[bend right] (7.75,-.25);
				\draw (2,.25) node{$1+\:_{0}z_{1}/2$};
				\draw (6,.25) node{$1+\:_{1}z_{1}/2$};
				\draw (4,-.75) node{$(1+\:_{0}z_{2}/2)^{2}$};
			\end{tikzpicture}
		\end{center}
	\end{figure}
	\vfill
	\begin{equation}
		(1+\:_{0}z_{1}/2)(1+\:_{1}z_{1}/2)\should(1+\:_{0}z_{2}/2)^{2} 
	\end{equation}
	{\small
	\begin{itemize}
		\item The current yield curve (${_{0}z_{1}},{_{0}z_{2}}$) predicts the future short-term yield ($_{1}z_{1}$), which we call the \underline{forward rate}
		\item This relationship between current yields and future yields is called the \underline{expectations hypothesis}
	\end{itemize}
	}
\end{frame}

\begin{frame}{\fttheory{Using the Yield Curve to Price Bonds}}
	Consider a $T$-year stream of semi-annual cash-flows.
	\begin{figure}[H]
		\begin{center}
			\begin{tikzpicture}
				\draw (0,0) -- (8,0);
				\foreach \x in {0,2,4,8}
				\draw(\x cm,3pt) -- (\x cm, -3pt);
				\draw (0,0) node[above=3pt] {0};
				\draw (2,0) node[above=3pt] {1};
				\draw (4,0) node[above=3pt] {2};
				\draw (8,0) node[above=3pt] {$2T$};
				\draw (0,0) node[below=3pt] {$-P$};
				\draw (2,0) node[below=3pt] {$c_{1}$};
				\draw (4,0) node[below=3pt] {$c_{2}$};
				\draw (6,0) node[above=3pt, align=center] {$\cdots$};
				\draw (6,0) node[below=3pt, align=center] {$\cdots$};
				\draw (8,0) node[below=3pt] {$c_{2T}$};
			\end{tikzpicture}
		\end{center}
	\end{figure}
	\noindent where $c_{1}$ is the six-month (0.5 year) cash-flow, $c_{2}$ twelve-month (1 year), $c_{3}$ eighteen-month (1.5 year), and so on
{\scriptsize
\begin{align*}
	P&\should\frac{c_{1}}{(1+z_{1}/2)}+\frac{c_{2}}{(1+z_{2}/2)^2}+\frac{c_{3}}{(1+z_{3}/2)^3}+\cdots+\frac{c_{2T}}{(1+z_{2T}/2)^{2T}}
\end{align*}
}
\noindent where $z_{1}$ is the six-month (0.5 year) zero-coupon yield (APR), $z_{2}$ twelve-month (1 year), $z_{3}$ eighteen-month (1.5 year), and so on
\end{frame}

\begin{frame}{\ftexampl{Using the Yield Curve to Price Bonds}}
	\small
	What is the \textbf{price} of a 2-year, 1000 USD 20\% coupon bond that makes semi-annual coupon payments. Suppose that the 6, 12, 18, and 24 month zero rates are 1\%, 3\%, 4\%, and 4.5\% respectively. What should the 2-year bond trade for?\\
	\pause
	{\color{blue}
		First, compute the present values:
		{\scriptsize
		\begin{align}
			\text{pv}_{1}&=\frac{100\text{ USD}}{(1+0.010/2)^{1}}=99.5025\text{ USD}\\
			\text{pv}_{2}&=\frac{100\text{ USD}}{(1+0.030/2)^{2}}=97.0662\text{ USD}\\
			\text{pv}_{3}&=\frac{100\text{ USD}}{(1+0.040/2)^{3}}=94.2322\text{ USD}\\
			\text{pv}_{4}&=\frac{100\text{ USD}}{(1+0.045/2)^{4}}=91.4843\text{ USD}
		\end{align}	
		}
		Next, note that the cash-flows from the zero coupon bond are 100 USD, 100 USD, 100 USD, and 1100 USD.
		{\scriptsize
		\begin{equation}
			\text{pv}_{\text{cpn}}=1\times \text{pv}_{1}+1\times \text{pv}_{2}+1\times \text{pv}_{3}+11\times \text{pv}_{4}=1297.1286\text{ USD}
		\end{equation}
		}
	}
\end{frame}

\begin{frame}{\fttheory{Yield-to-Maturity (YTM)}}
	\begin{itemize}
		\item Rather than carry around an entire yield curve whenever we want to recompute the price of a bond, we summarize the information with the \underline{yield-to-maturity} (YTM), with which you are familiar from your previous coursework
		\item Example: consider a 1000 USD, 30-year, 5\% T-bond that makes semi-annual payments, and trades for 1011.9610 USD 
		\item The YTM on this T-bond is the \underline{one} rate that would make the PV of cash-flows equal to the price, namely, 4.9233\%
	\end{itemize}
\end{frame}

	What is the \textbf{yield} of a 2-year, 1000 USD 20\% coupon bond that makes semi-annual coupon payments. Suppose that the 6, 12, 18, and 24 month zero rates are 1\%, 3\%, 4\%, and 4.5\% respectively. What should the 2-year bond trade for?\\

		\end{align*}
	}
\end{frame}

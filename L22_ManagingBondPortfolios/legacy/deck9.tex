% A PREVIEW-------------------------------------------------------------------%
\begin{frame}{Managing Bond Portfolios}
    \begin{itemize}
        \item Inverse relationship between bond prices and yields
        \item Important because interest rates fluctuate substantially. 
        \item Interest rate movements make investment risky 
        \item True even if the coupon and principal payments are guaranteed
    \end{itemize}
\end{frame}

% A PREVIEW-------------------------------------------------------------------%
\begin{frame}{Managing Bond Portfolios}
    \begin{itemize}
        \item Bond prices and yields are inversely related
        \item ...but the relationship is not linear (it is ``bowl shaped'')
        \item Different bonds have different sensitivities to interest rate changes
        \item Some bonds large price increase, others small when interest rates fall
    \end{itemize}
\end{frame}

% A PREVIEW-------------------------------------------------------------------%
\begin{frame}{Managing Bond Portfolios}
    \begin{itemize}
        \item Prices of long-term bonds tend to be more sensitive to interest rate changes than prices of short-term bonds; 
        \item As maturity increases, price sensitivity to yield changes increases at a decreasing rate;
        \item  Interest rate risk is inversely related to the bond’s coupon rate (low coupon bonds are more sensitive)
    \item  Bond prices are more sensitive to changes in yields when the bond is selling at a lower initial yield to maturity.
    \end{itemize}
\end{frame}

% DURATION--------------------------------------------------------------------%
\begin{frame}{Duration}
    \begin{itemize}
        \item Duration is a key concept in fixed income portfolio management. 
        \item  It is a simple summary statistic of the effective average maturity of the portfolio;
        \item  It is an essential tool in immunizing portfolio from interest rate risk;
        \item  It is a measure of the interest rate sensitivity of the portfolio.
    \end{itemize}
\end{frame}

% DURATION--------------------------------------------------------------------%
\begin{frame}{Duration}
\begin{itemize}
\item Cash flows $c_{1},c_{2},\ldots,c_{T}$ and yield $y$:
\begin{equation}
p(y)=\sum_{t=1}^{T}{\frac{c_{t}}{(1+y)^{t}}}
\end{equation}
\item The Macaulay's Duration is
    \begin{equation}
        D(y)=\sum_{t=1}^{T}{\frac{t}{p(y)}\cdot\frac{c_{t}}{(1+y)^{t}}}
    \end{equation}
\item The Modified Duration
    \begin{equation}
        D^{*}(y)=\frac{1}{1+y}\cdot\sum_{t=1}^{T}{\frac{t}{p(y)}\cdot\frac{c_{t}}{(1+y)^{t}}}
    \end{equation}
\end{itemize}
\end{frame}

% DURATION--------------------------------------------------------------------%
\begin{frame}{Duration}
\begin{itemize}
    \item Helpful:
        \begin{equation}
            \boxed{D^{*}(y)=\frac{D(y)}{1+y}}
        \end{equation}
    \item By the magic of calculus
\begin{equation}
    \frac{p'(y)}{p(y)}=-\frac{1}{1+y}\cdot\sum_{t=1}^{T}{\frac{t}{p(y)}\cdot\frac{c_{t}}{(1+y)^{t}}}=-D^{*}(y)
\end{equation}
\end{itemize}
\end{frame}


% DURATION--------------------------------------------------------------------%
\begin{frame}{Duration}
    \begin{itemize}
        \item Suppose that $y_{0}$ is the current yield of the bond
        \item We can approximate percent changes in the price 
        \begin{equation}
            \boxed{\frac{\Delta P}{P}\overset{A}{=}-D^{*}(y_{0})\cdot\Delta y}
        \end{equation}
    \end{itemize}
\end{frame}

% DURATION AND TERM-----------------------------------------------------------%
\begin{frame}{Duration and Term}
    \begin{itemize}
        \item Long-term bonds are more sensitive to interest rate movements than are short-term bonds.
        \item The duration measure enables us to quantify this relationship.
        \item When interest rates change, the proportional change in a bond’s price can be related to the change in its yield to maturity. 	 
    \end{itemize}
\end{frame}

% DURATION AND THE COUPON RATE------------------------------------------------%
\begin{frame}{Duration and the Coupon Rate}
    \begin{itemize}
        \item Holding maturity constant, a bond’s duration is higher when the coupon rate is lower;
        \item Holding the coupon rate constant, a bond’s duration generally increases with its time to maturity.
    \end{itemize}
\end{frame}

% CONVEXITY-------------------------------------------------------------------%
\begin{frame}{Convexity}
    \begin{itemize}
        \item $C^{*}$ denotes the convexity, or curvature
        \item Duration was a ``first-order'' approximation... 
        \item ...duration with convexity is a ``second-order'' approximation
            \begin{equation}
                \boxed{\frac{\Delta P}{P}\overset{A}{=}-D^{*}(y_{0})\Delta y+\tfrac{1}{2}C^{*}(y_{0})(\Delta y)^{2}}
            \end{equation}
        \item Calculus flashback: these are the first two terms of the Taylor expansion
    \end{itemize}
\end{frame}

% IMMUNIZATION----------------------------------------------------------------%
\begin{frame}{Immunization}
    \begin{itemize}
        \item Immunization techniques refer to strategies used by investors to shield their overall financial position from exposure to interest rate fluctuations. 
        \item If this is not taken into consideration, a change in interest rates could run havoc with the institution’s balance sheet, threatening bankruptcy.
    \end{itemize}
\end{frame}

% IMMUNIZATION----------------------------------------------------------------%
\begin{frame}{Immunization}
    \begin{itemize}
        \item Fixed income investors face two offsetting types of interest rate risks:
            Price risk; and  Reinvestment rate risk.
        \item Increases in interest rates cause capital losses but at the same time increase the rate at which reinvested income will grow.
        \item If the portfolio duration is chosen appropriately, these two effects will cancel out exactly. 
        \item When the portfolio’s duration is set equal to the investor’s horizon date, the accumulated value of the investment fund at the horizon date will be unaffected by interest rate fluctuations.
    \end{itemize}
\end{frame}

% IMMUNIZATION----------------------------------------------------------------%
\begin{frame}{Immunization}
    \begin{itemize}
        \item Even if a position is immunized, the portfolio manager still cannot feel safe. This is because of the need to rebalance the portfolio in response to changes in interest rates.
        \item In fact, as interest rates and asset durations change, the manager must rebalance the portfolio of fixed-interest instruments to align it with the asset side
    \end{itemize}
\end{frame}

% The Binomial Option Pricing Formula
\begin{frame}{The Binomial Option Pricing Formula}
\begin{itemize}
\item We'll price options by making the simplifying assumption that at the end of the next period, the stock price has only two possible values.\item The key insight of the Black-Scholes-Merton formula is that option payoffs can be replicated exactly by constructing a portfolio out of a risk-free bond and the underlying stock.
\item We'll begin by constructing a \textit{replicating portfolio}, a portfolio that has exactly the same payoffs as the option.
\item By the law of one price, the price of the option should be the same as the price of the replicating portfolio. 
\end{itemize}
\end{frame}

% A Two-State Single-Period Model
\begin{frame}{A Two-State Single-Period Model}
\begin{itemize}
\item Consider a European call option that expires in one period and has a strike price of \$50. 
\item In one period, the stock price will either rise by \$10 or fall by \$10. 
\item The one-period risk-free rate is 6\%. 
\end{itemize}

% A Two-State Single-Period Model
\begin{frame}{A Two-State Single-Period Model}
\begin{itemize}
\item We want to construct a portfolio consisting of the risk-free asset and the stock such that replicates the payoffs of the call. 
\item Let $B$ be the investment in the bonds and $\Delta$ be the number of shares of stock. 
\item We want to enforce two conditions:
\begin{align*}
&60\Delta+1.06B=10\tag{\text{Up}}\\
&40\Delta+1.06B=0\tag{\text{Down}}
\end{align*}
\item It turns out that the solution is $B=-18.87$ and $\Delta=0.5$ (we'll present a general formula in a moment).
\end{itemize}
\end{frame}

% The Binomial Pricing Formula
\begin{frame}{The Binomial Pricing Formula}
\begin{itemize}
\item Suppose the current stock price is $S$, and the stock price will either go up to $S_{u}$ or down to $S_{d}$.
\item The risk-free rate is $r_{F}$. 
\item The option has value $C_{u}$ if the stock goes up and $C_{d}$ if the stock goes down.
\end{itemize}

\begin{center}

% The Binomial Pricing Formula
\begin{frame}{The Binomial Pricing Formula}
\begin{itemize}
\item Again, we want to enforce two conditions:
\begin{align*}
&S_{u}\Delta+(1+r_{F})B=C_{u}\tag{\text{Up}}\\
&S_{d}\Delta+(1+r_{F})B=C_{d}\tag{\text{Down}}
\end{align*}
\item The solution is
\begin{align*}
&\Delta=\frac{C_{u}-C_{d}}{S_{u}-S_{d}}\\
&B=\frac{C_{d}-S_{d}\Delta}{1+r_{F}}
\end{align*}
\item \textbf{\color{magenta}{The option premium is}}
\begin{equation*}
C=\Delta\times S+B.
\end{equation*}
\item $\Delta$ \textbf{\color{magenta}{is the hedge ratio}}
\end{itemize}
\end{frame}

\frame[t]{\frametitle{Example}
Consider that today the Apple Inc stock price is \$90, and that you expect the price in one year to either be \$100 or \$80. The risk free rate is 10\%. Consider an european put option on Apple Inc that expires in one year with strike price of \$90. What is the price of the put option today? 
\pause
\begin{align}
100\times\Delta+100 \times (1+0.1) \times B=0\\
80\times\Delta+100 \times (1+0.1) \times B=10
\end{align}
Solve for $\Delta$ and $B$:
\begin{align}
\Delta=-0.5\\
B=0.4545
    \end{align}
\begin{align}
    P_{0}=-0.5\times90+0.4545\times 100=\$0.45
\end{align}
}

\begin{frame}{Risk Neutral Probabilities}
\begin{itemize}
    \item One important conclusion from last lecture's discussion is that investors are risk averse. As a result, we cannot price the the call option by just taking the present value of the expected cash flows.
    \item But can we correct the 'physical' probabilities and find the 'risk neutral' probabilities so that investors would be risk neutral? Answer: \textbf{Yes}.
\end{itemize}
\end{frame}

\begin{frame}{Risk Neutral Probabilities}
\begin{itemize}
    \item What is the advantage of know the risk neutral probability. It can save a lot of time. Because the price of the call option is:
    \begin{equation}
        C_0=\frac{p\times C_u +(1-p)\times C_d}{1+r}
    \end{equation}
where,
    \begin{equation}
        p=\frac{1+r-d}{u-d}
    \end{equation}
\item Important detail. If the stock price today is \$100 and in one year it can go up to \$125 or down to \$80, then $u=125/100=1.25$ and $d=80/100=0.8$.
\item If you use this approach, you should get the same result than the replicating portfolio approach.
\end{itemize}
\end{frame}

% The Black-Scholes-Merton Formula
\begin{frame}{The Black-Scholes-Merton Formula}
\begin{itemize}
\item The Black-Scholes-Merton formula for the European call is
\begin{equation*}
C=S\times N(d_{1})-PV(K)\times N(d_{2})
\end{equation*}
where
\begin{align*}
d_{1}&=\frac{\log(S/PV(K))}{\sigma\sqrt{T}}+\frac{\sigma\sqrt{T}}{2}\\
d_{2}&=d_{1}-\sigma\sqrt{T}
\end{align*}
\item $T$ is the time to expiration and $\sigma$ is the stock's volatility.
\item We learned about $N$ on the last slide. If you've forgotten how to use $\log$, Wiki it.
\item $\Delta=N(d_{1})$ and $B=-PV(K)\times N(d_{2})$. 
\end{itemize}
\end{frame}

% Risk and Return of an Option
\begin{frame}{Risk and Return of an Option}
\begin{itemize}
\item Let's compute the option beta. 
\item The simplest way is to compute the beta of the replicating portfolio.
\item The replicating portfolio has $S\Delta$ invested in the stock and $B$ invested in the risk-free asset.
\item Recall from our MM lecture:
\begin{align*}
\beta_{\text{option}}&=\left(\frac{S\Delta}{S\Delta+B}\right)\times\beta_{S}+\left(\frac{B}{S\Delta+B}\right)\times\beta_{r_{F}}\\
&=\left(\frac{S\Delta}{S\Delta+B}\right)\times\beta_{S}
\end{align*}
(remember that the beta of the risk-free asset is zero). 
\end{itemize}
\end{frame}

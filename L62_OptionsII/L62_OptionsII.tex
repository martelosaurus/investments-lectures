% The B_{c}inomial Option Pricing Formula
\begin{frame}{\ftheader{Option Pricing}}
	\begin{itemize}
		\item The key insight of option pricing theory is that option payoffs can be replicated exactly by constructing a portfolio out of the risk-free asset and the underlying stock
		\item We'll use this insight to price and hedge options
		\item To make things simple, we are going to assume that at maturity, the stock price can take on only two possible values (this is called digital or binomial option pricing)
		\item Note: take the risk-free rate to be 10bps in the examples
	\end{itemize}
\end{frame}

%-----------------------------------------------------------------------------%
% binomial options: calls
\begin{frame}{\fttheory{Binomial Options: Calls}}
	$S_{0}$ is the spot price on date $0$. On date $1$, the price is either $S_{u}$ (up) or $S_{d}$ (down), $K$ is the strike price, and $C_{0}$ is the premium
    \begin{center}
        \begin{tabular}{lcccc}
			Security & Notes & CF$_{0}$ & CF$_{u}$ & CF$_{d}$ \\ \hline\hline
			Call & Long & $-C_{0}$ & $S_{u}-K$ &$0$\\ \hline\hline
			Stock & Long $\Delta_{c}$  & $-\Delta_{c} S_{0}$ & $\Delta_{c} S_{u}$ & $\Delta_{c} S_{d}$ \\
			Risk-Free & Borrow & $B_{c}$ & $-(1+r_{F})B_{c}$ & $-(1+r_{F})B_{c}$\\ \hline
			& & $-\Delta_{c} S_{0}+B_{c}$ & $S_{u}-K$ & $0$ \\ \hline\hline
        \end{tabular}
    \end{center}
	\begin{equation}
		\boxed{\Delta_{c}=\frac{S_{u}-K}{S_{u}-S_{d}},\:B_{c}=\frac{\Delta_{c}S_{d}}{1+r_{F}},\:C_{0}=\Delta_{c}S_{0}-B_{c}}
	\end{equation}
	$\Rightarrow$ To hedge long call, short $\Delta_{c}$ of stock and save $B_{c}$.
\end{frame}

% binomial options: put
\begin{frame}{\fttheory{Binomial Options: Put}}
	$S_{0}$ is the spot price on date $0$. On date $1$, the price is either $S_{u}$ (up) or $S_{d}$ (down), $K$ is the strike price, and $P_{0}$ is the premium
    \begin{center}
        \begin{tabular}{lcccc}
			Security & Notes & CF$_{0}$ & CF$_{u}$ & CF$_{d}$ \\ \hline\hline
			Put & Long & $-P_{0}$ & $0$ &$K-S_{d}$\\ \hline\hline
			Stock & Short $\Delta_{p}$ & $\Delta_{p} S_{0}$ & $-\Delta_{p} S_{u}$ & $-\Delta_{p} S_{d}$ \\
			Risk-Free & Save & $-B_{p}$ & $(1+r_{F})B_{p}$ & $(1+r_{F})B_{p}$\\ \hline
			& & $\Delta_{p} S_{0}-B_{p}$ & $0$ & $K-S_{d}$ \\ \hline\hline
        \end{tabular}
    \end{center}
	\begin{equation}
		\boxed{\Delta_{p}=\frac{K-S_{d}}{S_{u}-S_{d}},\:B_{p}=\frac{\Delta_{p}S_{u}}{1+r_{F}},\:P_{0}=B_{p}-\Delta_{p}S_{0}}
	\end{equation}
	$\Rightarrow$ To hedge long put, borrow $B_{p}$ and long $\Delta_{p}$ of stock 
\end{frame}

% factors affeting option prices
\begin{frame}{\fttheory{Factors Affecting Option Prices}}
	\begin{itemize}
		\item \textbf{Strike Price and Stock Price:} the value of a call is \underline{decreasing} in the strike price (the higher the strike, the lower the value); the value of a call is \underline{increasing} in the price of the stock (the higher the price of the stock, the higher the value). 
		\item \textbf{Arbitrage Bounds on Option Prices:} 
		\begin{itemize}
			\item an American option carries all of the same rights and privileges as an otherwise equivalent European option. The American option can't be worth less. 
			\item a put option can't be worth more than its strike price and a call can't be worth more than the stock itself.
		\end{itemize}
	\end{itemize}
\end{frame}

%-----------------------------------------------------------------------------%
% factors affeting option prices
\begin{frame}{\fttheory{Factors Affecting Option Prices}}
	\begin{itemize}
		\item \textbf{Option Prices and the Exercise Date:} 
		\begin{itemize}
			\item (for American options,) the longer the time to the exercise date, the more valuable the option.
			\item An American option with a later exercise date cannot be worth less than an otherwise identical American option with an earlier exercise date.
			\item These arguments \underline{do not} carry through for European options.
		\end{itemize}
		\item \textbf{Option Prices and Volatility:} 
		\begin{itemize}
			\item The value of an option generally increases with the volatility of the stock.
			\item Increasing the volatility of the stock increases the likelihood that the stock will take on extreme values (which is where options are valuable).
		\end{itemize}
	\end{itemize}
\end{frame}

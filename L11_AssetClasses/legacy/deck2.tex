%=============================================================================%
%   THE MONEY MARKET                                                          % 
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item The money market is a subset of the bond market.
\item These are short-term, highly liquid securities.
\item They are often called \textit{cash} or \textit{cash equivalents}.
\end{itemize}
\end{frame}

%=============================================================================%
%   THE MONEY MARKET                                                          %
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item[1.]\textbf{Treasury Bills.} 
\begin{itemize}
\item This is the simplest money market instrument.
\item The US government sells bills to raise money.
\item Maturities of 4, 13, 26, and 52 weeks. 
\item Not to be confused with Treasury Notes and Bonds. 
\item Considered ``risk-free''.
\end{itemize}
\item[2.]\textbf{Certificates of Deposits.} 
\begin{itemize}
\item Certificates of Deposit (CDs) are time deposits with a bank 
\item Unlike a checking account, you cannot withdraw funds on demand
\item They are insured by the FDIC and may be traded on a secondary market
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE MONEY MARKET                                                          %
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item[3.]\textbf{Commercial Paper.} 
\begin{itemize}
\item Large corporations issue short-term notes to finance working capital
\item They have maturities less than 270 days to avoid registration with the SEC
\item They were historically used by non-financial firms, but notoriously used by financial firms just before the crisis  
\end{itemize}
\item[4.]\textbf{Bankers' Acceptances.} 
\begin{itemize}
\item Purchaser authorizes a bank to pay a seller for goods at later date (time draft)
\item When purchaser’s bank “accepts” draft, it becomes contingent liability of the bank (and marketable)
\item Often used to back international sales of goods and services
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE MONEY MARKET                                                          %
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item[5.]\textbf{Eurodollars.} 
\begin{itemize}
\item Dollar-denominated time deposits held outside U.S.
\item Pay higher interest rate than U.S. deposits due to liquidity risks
\end{itemize}
\item[6.]\textbf{Repos.} 
\begin{itemize}
\item Short-term sales of securities with promise to repurchase at higher price
\item RP is a collateralized loan
\item Many RPs are overnight; ``Term'' RPs may have a 1-month maturity
\end{itemize}
\item[7.]\textbf{Reverse Repos.}
\begin{itemize}
\item Lending money; obtaining security title as collateral
\item ``Haircuts'' may be required depending on collateral
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE MONEY MARKET                                                          %
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item[8.]\textbf{Brokers' Calls.} 
\begin{itemize}
\item Call money rate applies for investors buying stock on margin
\item Loan may be ``called in'' by broker
\end{itemize}
\item[9.]\textbf{Federal Funds.} 
\begin{itemize}
\item Trading in reserves held at the Federal Reserve
\item Key interest rate for economy
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE MONEY MARKET                                                          %
%=============================================================================%
\begin{frame}{The Money Market}
\begin{itemize}
\item[10.] LIBOR (London Interbank Offered Rate)
\begin{itemize}
\item Second only to Treasuries in importance. 
\item It is the rate at which banks lend to each other.
\item It is used as a benchmark rate for trillions of dollars worth of securities (primarily derivatives). 
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   LIBOR SCANDAL                                                             %
%=============================================================================%
\begin{frame}{The Libor Scandal}
\begin{center}
\includegraphics[scale=.25]{libor}
\end{center}
\begin{itemize}
\item Before 2012, LIBOR was literally a poll.
\item Banks submitted what they ``thought'' they could borrow/lend at with other banks. 
\item It was discovered that banks were manipulating the poll to favor their trading positions.
\end{itemize}
\end{frame}

%=============================================================================%
%   THE BOND MARKET                                                           %
%=============================================================================%
\begin{frame}{The Bond Market}
\begin{itemize}
\item[1.]\textbf{Treasury Notes and Bills.} 
\begin{itemize}
\item Notes: up to 10 year maturities
\item Bills: up to 30 year maturities
\end{itemize}
\item[2.]\textbf{Treasury Inflation Protected Securities (TIPS).} 
\begin{itemize}
\item Principal adjusted for changes in the CPI
\item Marked with a trailing ``i'' in quote sheets
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE BOND MARKET                                                           %
%=============================================================================%
\begin{frame}{The Bond Market}
\begin{itemize}
\item[3.]\textbf{Federal Agency Debt.} 
\begin{itemize}
\item Most are home-mortgage-related: FNMA (``Fannie''), FHLMC (''Freddi''), GNMA (``Ginnie''), Federal Home Loan Banks
\item Risks of these securities?
\begin{itemize}
\item Implied backing by the government
\item In September 2008, federal government took over Fannie and Freddie
\end{itemize}
\end{itemize}
\item[4.]\textbf{International Bonds.} 
\begin{itemize}
\item Bonds issued in another currency
\item E.g. Yankee bond is a bond sold in the US in USD by a non-US issuer; Samuari bond is a bond sold in Japan in Yen by a non-Japanese issuer, etc. 
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   THE BOND MARKET                                                           %
%=============================================================================%
\begin{frame}{The Bond Market}
\begin{itemize}
\item[5.]\textbf{Municipal Bonds.} 
\begin{itemize}
\item Issued by state and local governments to finance capital improvement projects (e.g. roads, schools, parks). 
\item Interest income is exempt from state and local taxation.
\end{itemize}
\item[6.]\textbf{Corporate Bonds.} 
\begin{itemize}
\item You know a lot about these already. 
\item Two new species:
\begin{itemize}
\item\textbf{Callable Bonds:} The firm can buy back the bond at a specified price.
\item\textbf{Convertible Bonds:} The bondholder can convert her bond into a specified number of shares of common stock.
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{The Bond Market}
\begin{itemize}
\item[7.]\textbf{Asset Backed Securities (ABSs).} 
\begin{itemize}
\item Put a bunch of assets in a portfolio called a ``pool.''
\item Create securities out of shares in the pool.
\item Voila! You've made an asset backed security.
\item Examples of assets turned into ABSs:
\begin{itemize}
\item Mortgages (``Mortgage Backed Securities'')
\item Life insurance policies (``death bonds'')
\item Credit card debt
\item Student debt
\item Receivables
\item Home equity loans
\item Auto loans
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

%=============================================================================%
%   INDICES                                                                   %
%=============================================================================%
\begin{frame}{Indices}
\begin{itemize}
\item Quick math review.
\item Suppose we have $N$ numbers, $p_{1},p_{1},\ldots,p_{N}$.
\item We'll write
\begin{equation}
\sum_{j=1}^{N}{p_{j}}=p_{1}+p_{2}+\cdots+p_{N}
\end{equation}
\end{itemize}
\end{frame}

%=============================================================================%
%   INDICES                                                                   %
%=============================================================================%
\begin{frame}{Price-Weighted (E.g. the DIJA)}
\begin{itemize}
\item $N$ stocks; 
\item $p_{j}$ price of stock $j$;
\item The \textit{value} of the \textbf{price-weighted index} is
\begin{equation}
\frac{1}{N}\sum_{j=1}^{N}{p_{j}}
\end{equation}
\end{itemize}
\textbf{Example:}
\begin{center}
\begin{tabular}{cc}
\textbf{Company} & \textbf{Price} \\ \hline
$A$ & \$6 \\
$B$ & \$20
\end{tabular}
\end{center}
\begin{center}
What's the \textit{value} of the index?
\end{center}
\end{frame}

%=============================================================================%
%   INDICES                                                                   %
%=============================================================================%
\begin{frame}{Equally-Weighted}
\begin{itemize}
\item $N$ stocks; 
\item $p_{j,t}$ price of stock $j$ on day $t$; 
\item The \textit{return} of an \textbf{equally-weighted index} is
\begin{equation}
\frac{1}{N}\sum_{j=1}^{N}{\frac{p_{j,t+1}-p_{j,t}}{p_{j,t}}}
\end{equation}
\end{itemize}
\textbf{Example:}
\begin{center}
\begin{tabular}{ccc}
\textbf{Company} & \textbf{Price$_{t}$} & \textbf{Price$_{t+1}$} \\ \hline
$A$ & \$4 & \$3 \\
$B$ & \$10 & \$11
\end{tabular}
\end{center}
\begin{center}
What's the \textit{return} of the index?
\end{center}
\end{frame}

%=============================================================================%
%   INDICES                                                                   %
%=============================================================================%
\begin{frame}{Indices}
\begin{itemize}
\item Let's get fancy. Let's have $N$ \textit{weights} $w_{1},w_{2},\ldots,w_{N}$. 
\item The \textit{weighted} sum of $p_{1},p_{1},\ldots,p_{N}$ is
\begin{equation}
\sum_{j=1}^{N}{w_{j}p_{j}}=w_{1}p_{1}+w_{2}p_{2}+\cdots+w_{N}p_{N}.
\end{equation}
\end{itemize}
\textbf{Example:}
\begin{center}
\begin{tabular}{cccc}
\textbf{Company} & \textbf{Weight} & \textbf{Price} \\ \hline
$A$ & 25 & \$8 \\
$B$ & 75 & \$32
\end{tabular}
\end{center}
\begin{center}
What's the weighted sum? 
\end{center}
\end{frame}

%=============================================================================%
%   INDICES                                                                   %
%=============================================================================%
\begin{frame}{Value-Weighted (E.g. S\&P 500)}
\begin{itemize}
\item $N$ stocks; 
\item $p_{j,t}$ price of stock $j$ on day $t$; 
\item $n_{j}$ number of shares of stock $j$. 
\item The \textit{value} of the \textbf{value-weighted index} is
\begin{equation}
\sum_{j=1}^{N}{n_{j}p_{j,t}}.
\end{equation}
\item You can compute the \textit{return} from day $t$ to day $t+1$: 
\begin{equation}
\frac{\sum_{j=1}^{N}{n_{j}p_{j,t+1}}}{\sum_{j=1}^{N}{n_{j}p_{j,t}}}-1
\end{equation}
\end{itemize}
\end{frame}
